
DROP DATABASE IF EXISTS customer;
CREATE DATABASE customer;
USE customer;

create table CARDLEVELS
(
  CARDLEVELS_ID        INTEGER  not null AUTO_INCREMENT  comment '等级编号',
  CARDLEVELS_LEVELNAME VARCHAR(50) comment '等级名称',
  CARDLEVELS_NEEDPOINT INTEGER(7) comment '所需最大积分',
  CARDLEVELS_POINT     FLOAT(3,2) comment '(兑换)积分比例',
  CARDLEVELS_PERCENT   FLOAT(3,2) comment '折扣比例',
  CONSTRAINT  PK_CARDLEVELS primary key (CARDLEVELS_ID)
)ENGINE=InnoDB DEFAULT CHARACTER SET=utf8 COLLATE=utf8_unicode_ci COMMENT='会员等级';

create table CATEGORIES
(
  CATEGORY_ID    INTEGER not null AUTO_INCREMENT  comment '等级编号',
  CATEGORY_NAME  VARCHAR(20) comment '类别说明',
  CATEGORY_TITLE VARCHAR(20) comment '类别标题',
  CATEGORY_PID   INTEGER comment '类别父ID',
  CATEGORY_ISDEL INTEGER(1) comment '类别是否删除 1 删除 0 正常',
  constraint PK_CATEGORIES primary key (CATEGORY_ID)
)ENGINE=InnoDB DEFAULT CHARACTER SET=utf8 COLLATE=utf8_unicode_ci COMMENT='类别';

create table SHOPS
(
  SHOP_ID            INTEGER not null AUTO_INCREMENT  comment '店铺编号',
  SHOP_NAME          VARCHAR(20) comment '店铺名称',
  SHOP_CATEGORY_ID   INTEGER comment '类别ID',
  SHOP_CONTACTNAME   VARCHAR(20) comment '联系人',
  SHOP_CONTACTTEL    VARCHAR(20) comment '联系电话',
  SHOP_ADDRESS       VARCHAR(50) comment '地址',
  SHOP_REMARK        VARCHAR(100) comment '备注',
  SHOP_ISHASSETADMIN INTEGER(1) comment '是否已分配管理员',
  SHOP_CREATETIME    DATETIME comment '加盟时间',
  SHOP_ISDEL         INTEGER(1) comment '店铺是否删除 1 删除 0正常',
  constraint PK_SHOPS primary key (SHOP_ID),
  constraint FK_SHOPS_RELATIONS_CATEGORI foreign key (SHOP_CATEGORY_ID)  references CATEGORIES (CATEGORY_ID)
)ENGINE=InnoDB DEFAULT CHARACTER SET=utf8 COLLATE=utf8_unicode_ci COMMENT='店铺';

create table EXCHANGGIFTS
(
  EXCHANGGIFTS_ID         INTEGER not null AUTO_INCREMENT  comment '礼品编号',
  SHOP_ID                 INTEGER comment '店铺编号',
  EXCHANGGIFTS_GIFTCODE   VARCHAR(255) comment '礼品编码',
  EXCHANGGIFTS_GIFTNAME   VARCHAR(255) comment '礼品名称',
  EXCHANGGIFTS_PHOTO      VARCHAR(255) comment '礼品图片',
  EXCHANGGIFTS_POINT      INTEGER(5) comment '所需积分',
  EXCHANGGIFTS_INTEGER     INTEGER(5) comment '总数量',
  EXCHANGGIFTS_EXCHANGNUM INTEGER(5) comment '已兑换的数量',
  EXCHANGGIFTS_REMARK     VARCHAR(255) comment '联系人',
  EXCHANGGIFTS_STATE      INTEGER(1) comment '礼品状态',
  EXCHANGGIFTS_ISDEL      INTEGER(1) comment '礼品是否删除 1 删除 0正常',
  constraint PK_EXCHANGGIFTS primary key (EXCHANGGIFTS_ID),
  constraint FK_EXCHANGG_RELATIONS_SHOPS foreign key (SHOP_ID) references SHOPS (SHOP_ID)
)ENGINE=InnoDB DEFAULT CHARACTER SET=utf8 COLLATE=utf8_unicode_ci COMMENT='礼品';

create table MEMCARDS
(
  MEMCARDS_ID             INTEGER not null AUTO_INCREMENT  comment '会员编号',
  CARDLEVELS_ID           INTEGER comment '等级编号',
  SHOP_ID                 INTEGER comment '店铺编号',
  MEMCARDS_CARDID         varchar(50) comment '会员卡号',
  MEMCARDS_PASSWORD       varchar(20) comment '卡片密码',
  MEMCARDS_NAME           varchar(20) comment '会员姓名',
  MEMCARDS_SEX            varchar(6) comment '会员性别',
  MEMCARDS_MOBILE         varchar(50) comment '手机号码',
  MEMCARDS_PHOTO          varchar(200) comment '靓照',
  MEMCARDS_BIRTHDAY_MONTH INTEGER(2) comment '会员生日--月',
  MEMCARDS_BIRTHDAY_DAY   INTEGER(2) comment '会员生日--日',
  MEMCARDS_BIRTHDAYTYPE   INTEGER(1) comment '会员生日类型',
  MEMCARDS_ISPAST         INTEGER(1) comment '是否设置卡片过期时间',
  MEMCARDS_PASTTIME       DATE comment '卡片过期时间',
  MEMCARDS_POINT          INTEGER(7) comment '当前积分',
  MEMCARDS_MONEY          FLOAT(7,2) comment '卡片付费',
  MEMCARDS_TOTALMONEY     FLOAT(10,2) comment '累计消费',
  MEMCARDS_TOTALCOUNT     INTEGER(5) comment '累计消费次数',
  MEMCARDS_STATE          INTEGER comment '卡片状态',
  MEMCARDS_ISPOINTAUTO    INTEGER(1) comment '积分是否可以自动换成等级',
  MEMCARDS_REFERERID      INTEGER comment '推荐人ID',
  MEMCARDS_REFERERCARD    varchar(50) comment '推荐人卡号',
  MEMCARDS_REFERERNAME    varchar(20) comment '推荐人姓名',
  MEMCARDS_CREATETIME     DATETIME comment '登记日期',
  MEMCARDS_ISDEL          INTEGER(1) comment '会员是否删除 1 删除 0正常',
  constraint PK_MEMCARDS primary key (MEMCARDS_ID),
  constraint FK_MEMCARDS_RELATIONS_CARDLEVE foreign key (CARDLEVELS_ID)  references CARDLEVELS (CARDLEVELS_ID),
  constraint FK_MEMCARDS_RELATIONS_CATEGORI foreign key (MEMCARDS_STATE) references CATEGORIES (CATEGORY_ID),
  constraint FK_MEMCARDS_RELATIONS_SHOPS foreign key (SHOP_ID) references SHOPS (SHOP_ID)
)ENGINE=InnoDB DEFAULT CHARACTER SET=utf8 COLLATE=utf8_unicode_ci COMMENT='会员';

create table USERS
(
  USERS_ID        INTEGER not null AUTO_INCREMENT  comment '用户编号',
  SHOP_ID         INTEGER comment '店铺编号',
  USERS_LOGINNAME VARCHAR(20) comment '用户登录名',
  USERS_PASSWORD  varchar(255) comment '密码',
  USERS_REALNAME  varchar(20) comment '真实姓名',
  USERS_SEX       varchar(6) comment '性别',
  USERS_TELEPHONE varchar(20) comment '电话',
  USERS_ROLE      INTEGER comment '角色',

  constraint FK_USERS_RELATIONS_CATEGORI foreign key (USERS_ROLE) references CATEGORIES (CATEGORY_ID),
